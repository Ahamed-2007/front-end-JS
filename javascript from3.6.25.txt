<script>
    function calculate() {
        let m1 = parseFloat(document.getElementById("m1").value);
        let m2 = parseFloat(document.getElementById("m2").value);
        let operation = document.getElementById("o1").value.trim();
        let result;
            switch (operation) {
                case '+':
                    result = m1 + m2;
                    break;
                case '-':
                    result = m1 - m2;
                    break;
                case '*':
                    result = m1 * m2;
                    break;
                case '/':
                    result = m2 !== 0 ? m1 / m2 : "Cannot divide by zero.";
                    break;
                default:
                    result = "Invalid Operation";
            }
        }

        document.getElementById("p1").innerHTML = "Answer Result: " + result;
    }
</script>
